// Función para centrar un bloque completo function centerBlock(blockIds) { blockIds.forEach(id=>{ const el = document.getElementById(id); el.style.left = '50%'; el.style.transform = 'translateX(-50%) translateY(0)'; }); } // Función para resetear al estado original (alineado a la G) function resetBlock(blockIds) { blockIds.forEach(id=>{ const el = document.getElementById(id); if(id.includes('overlay')) { el.style.transform = 'translateX(-50%) translateY(20px)'; } else { el.style.transform = 'translateX(0) translateY(0)'; } if(id === 'overlay-top' || id === 'overlay-bottom') { el.style.left = '50%'; } else { el.style.left = '0.09em'; } }); } // Observers para bloques superior e inferior const blockTop = ['overlay-top','museo','fia-superior']; const blockBottom = ['overlay-bottom','esceno','fia-inferior']; const observerTop = new IntersectionObserver(entries=>{ entries.forEach(entry=>{ if(entry.isIntersecting){ // Mantener animaciones PA existentes setTimeout(()=> centerBlock(blockTop), 800); } else { resetBlock(blockTop); } }); },{threshold:0.5}); const observerBottom = new IntersectionObserver(entries=>{ entries.forEach(entry=>{ if(entry.isIntersecting){ setTimeout(()=> centerBlock(blockBottom), 800); } else { resetBlock(blockBottom); } }); },{threshold:0.5}); observerTop.observe(document.getElementById('museo')); observerBottom.observe(document.getElementById('esceno'));
